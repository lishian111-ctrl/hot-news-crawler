# 能源行业信息情报网站实施计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 构建一个本地部署的能源行业信息情报网站，实现油气行业、海上风电、FFML 三个板块的信息自动采集、展示、热点筛选和收藏功能。

**Architecture:**
- 前后端分离架构，前端 Vue 3 + Element Plus，后端 FastAPI RESTful API
- 爬虫模块独立部署，支持 Playwright（动态网站）和 BeautifulSoup（静态网站）
- SQLite 数据库存储新闻、信源、收藏数据
- Windows Task Scheduler + APScheduler 实现每天 09:00 定时采集

**Tech Stack:**
- 前端：Vue 3、Element Plus、Vite、Vue Router、Axios
- 后端：Python 3.10+、FastAPI、uvicorn、SQLite、SQLAlchemy
- 爬虫：Playwright、BeautifulSoup4、lxml
- 工具：pandas（Excel 处理）、openpyxl、 APScheduler

- [ ] **Step 2: 创建 base.py（爬虫基类）**

```python
from abc import ABC, abstractmethod
from typing import Dict, List, Optional
from datetime import datetime
import hashlib

class CrawlResult:
    """爬取结果数据类"""
    def __init__(self, title: str, content: str, url: str,
                 publish_time: Optional[datetime] = None,
                 source_name: str = ""):
        self.title = title
        self.content = content
        self.url = url
        self.publish_time = publish_time
        self.source_name = source_name
        self.summary = self._generate_summary()
        self.content_hash = self._generate_hash()

    def _generate_summary(self) -> str:
        """生成摘要（前 200 字）"""
        text = self.content[:200] + "..." if len(self.content) > 200 else self.content
        return text.replace("\n", " ").strip()

    def _generate_hash(self) -> str:
        """生成内容哈希（用于去重）"""
        content = f"{self.title}{self.content}"
        return hashlib.sha256(content.encode()).hexdigest()

class BaseCrawler(ABC):
    """爬虫基类"""

    def __init__(self, timeout: int = 30):
        self.timeout = timeout

    @abstractmethod
    def crawl(self, url: str) -> List[CrawlResult]:
        """
        爬取指定 URL 的内容
        返回：CrawlResult 列表
        """
        pass

    @abstractmethod
    def get_list_page_urls(self, base_url: str) -> List[str]:
        """
        获取列表页 URL
        返回：URL 列表
        """
        pass

    def parse_html(self, html: str) -> Dict:
        """
        解析 HTML 内容（子类实现具体选择器）
        """
        pass
```

- [ ] **Step 3: 创建 spider.py（通用爬虫）**

```python
from crawler.base import BaseCrawler, CrawlResult
from typing import List, Dict, Optional
from datetime import datetime
import requests
from bs4 import BeautifulSoup
from loguru import logger
import time

class GeneralSpider(BaseCrawler):
    """通用爬虫 - 支持 CSS 选择器配置"""

    def __init__(self, rules: Dict, timeout: int = 30):
        super().__init__(timeout)
        self.rules = rules
        self.session = requests.Session()
        self.session.headers.update({
            "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36"
        })

    def crawl(self, url: str) -> List[CrawlResult]:
        """爬取单个 URL 内容"""
        results = []
        try:
            response = self.session.get(url, timeout=self.timeout)
            response.encoding = response.apparent_encoding
            html = response.text

            title = self._extract(html, self.rules.get("title_selector", ""))
            content = self._extract(html, self.rules.get("content_selector", ""), html=True)
            publish_time = self._extract_time(html, self.rules.get("time_selector", ""))

            if title and content:
                results.append(CrawlResult(
                    title=title.strip(),
                    content=self._clean_content(content),
                    url=url,
                    publish_time=publish_time,
                    source_name=self.rules.get("source_name", "")
                ))
        except Exception as e:
            logger.error(f"爬取失败 {url}: {str(e)}")

        return results

    def get_list_page_urls(self, base_url: str) -> List[str]:
        """获取列表页 URL（简单实现，返回单个 URL）"""
        return [base_url]

    def _extract(self, html: str, selector: str, html: bool = False) -> str:
        """提取元素文本"""
        if not selector:
            return ""
        soup = BeautifulSoup(html, "lxml")
        element = soup.select_one(selector)
        if element:
            return element.get_text(strip=True) if not html else str(element)
        return ""

    def _extract_time(self, html: str, selector: str) -> Optional[datetime]:
        """提取时间"""
        time_str = self._extract(html, selector)
        if not time_str:
            return None

        # 尝试多种时间格式
        formats = [
            "%Y-%m-%d %H:%M:%S",
            "%Y-%m-%d",
            "%Y/%m/%d",
            "%Y年%m月%d日",
            "%Y-%m-%dT%H:%M:%S"
        ]

        for fmt in formats:
            try:
                return datetime.strptime(time_str.strip(), fmt)
            except ValueError:
                continue
        return None

    def _clean_content(self, content: str) -> str:
        """清洗 HTML 内容，保留基本格式"""
        soup = BeautifulSoup(content, "lxml")
        # 移除脚本和样式
        for tag in soup(["script", "style"]):
            tag.decompose()
        # 保留段落
        return str(soup)
```

```python
from typing import List, Optional
from sqlalchemy.orm import Session
from models.news import News
import difflib

class DedupService:
    """新闻去重服务"""

    def __init__(self, db: Session):
        self.db = db
        self.title_threshold = 0.8  # 标题相似度阈值
        self.content_threshold = 0.7  # 内容相似度阈值

    def is_duplicate(self, title: str, content: str, url: str = None) -> bool:
        """
        检查是否重复
        返回：True=重复，False=不重复
        """
        # 1. URL 精确匹配（最快）
        if url:
            existing = self.db.query(News).filter(News.source_url == url).first()
            if existing:
                return True

        # 2. 标题相似度匹配
        recent_news = self.db.query(News).order_by(News.created_at.desc()).limit(100).all()
        for news in recent_news:
            # 标题相似度
            title_sim = difflib.SequenceMatcher(None, title, news.title).ratio()
            if title_sim >= self.title_threshold:
                return True

        return False

    def find_similar(self, title: str, limit: int = 5) -> List[News]:
        """查找相似新闻（用于人工判断）"""
        recent_news = self.db.query(News).order_by(News.created_at.desc()).limit(50).all()
        similar = []

        for news in recent_news:
            title_sim = difflib.SequenceMatcher(None, title, news.title).ratio()
            if title_sim >= 0.5:
                similar.append((news, title_sim))

        similar.sort(key=lambda x: x[1], reverse=True)
        return [news for news, _ in similar[:limit]]
```

- [ ] **Step 3: 创建 scoring.py（打分服务）**

```python
from typing import Dict, List
from models.news import News
from models.source import Source
from datetime import datetime, timedelta
from config import (
    SCORE_KEYWORDS, SCORE_KEYWORDS_POINTS,
    CONTENT_TYPE_KEYWORDS, CONTENT_TYPE_POINTS
)

class ScoringService:
    """热点打分服务"""

    def calculate_score(self, news: News, source: Source = None) -> int:
        """
        计算新闻热点分数
        总分 = 时效性 (25) + 信源权重 (30) + 关键词 (25) + 内容类型 (20)
        """
        score = 0

        # 1. 时效性打分 (0-25 分)
        score += self._time_score(news.publish_time)

        # 2. 信源权重打分 (0-30 分)
        if source:
            score += self._source_weight_score(source.weight)

        # 3. 关键词匹配打分 (0-25 分)
        score += self._keyword_score(news.title, news.content)

        # 4. 内容类型打分 (0-20 分)
        score += self._content_type_score(news.title, news.content)

        return score

    def _time_score(self, publish_time: datetime) -> int:
        """时效性打分"""
        if not publish_time:
            return 5

        now = datetime.now()
        delta = now - publish_time

        if delta < timedelta(hours=2):
            return 25
        elif delta < timedelta(hours=4):
            return 20
        elif delta < timedelta(hours=8):
            return 15
        elif delta < timedelta(hours=24):
            return 10
        else:
            return 5

    def _source_weight_score(self, weight: int) -> int:
        """信源权重打分"""
        return min(weight * 3, 30)  # 权重×3，最高 30 分

    def _keyword_score(self, title: str, content: str) -> int:
        """关键词匹配打分"""
        text = f"{title} {content}".lower()

        # 按优先级检查（从高到低）
        for category, keywords in SCORE_KEYWORDS.items():
            for keyword in keywords:
                if keyword.lower() in text:
                    return SCORE_KEYWORDS_POINTS.get(category, 15)

        return 15  # 默认一般词分数

    def _content_type_score(self, title: str, content: str) -> int:
        """内容类型打分"""
        text = f"{title} {content}".lower()

        # 按优先级检查（从高到低）
        for content_type, keywords in CONTENT_TYPE_KEYWORDS.items():
            for keyword in keywords:
                if keyword.lower() in text:
                    return CONTENT_TYPE_POINTS.get(content_type, 8)

        return CONTENT_TYPE_POINTS["一般资讯"]
```

```python
            df = pd.read_excel(filepath)

            columns_map = {

                "分类": "category",

            }
            df = df.rename(columns=columns_map)

            df["board"] = board

            # 默认权重
            if "weight" not in df.columns:
                df["weight"] = 5

            return df.to_dict("records")
```

```python
    logger.info(f"加载 {len(sources)} 条信源")
    return sources

def crawl_source(source: Source, rules: dict) -> List[News]:
    """爬取单个信源"""
    results = []

    # 获取匹配的规则
    rule = rules.get(source.name, {})
    if not rule:
        logger.warning(f"未找到 {source.name} 的采集规则")
        return results

    spider = GeneralSpider({"rules": rule})

    try:
        crawl_results = spider.crawl(source.url)
        for result in crawl_results:
            results.append({
                "title": result.title,
                "content": result.content,
                "summary": result.summary,
                "source_url": result.url,
                "source_id": source.id,
                "category": source.board,
                "publish_time": result.publish_time,
                "content_hash": result.content_hash
            })
    except Exception as e:
        logger.error(f"爬取 {source.name} 失败：{e}")

    return results

def run_crawl_task():
    """执行采集任务"""
    logger.info("=" * 50)
    logger.info("开始采集任务")
    logger.info("=" * 50)

    init_database()
    db = SessionLocal()
    dedup = DedupService(db)
    scoring = ScoringService()

    # 加载信源
    sources = db.query(Source).all()
    if not sources:
        sources = load_sources_from_excel()

    # 获取规则映射
    rules_mapping = {
        "oil_gas": OIL_GAS_RULES,
        "wind_power": WIND_RULES,
        "ffml": FFML_RULES
    }

    total_new = 0

    for source in sources:
        logger.info(f"正在采集：{source.name}")

        rules = rules_mapping.get(source.board, {})
        news_items = crawl_source(source, rules)

        for item in news_items:
            # 去重检查
            if dedup.is_duplicate(item["title"], item["content"], item["source_url"]):
                logger.debug(f"跳过重复：{item['title'][:30]}...")
                continue

            # 创建新闻对象
            news = News(**item)

            # 打分
            news.score = scoring.calculate_score(news, source)

            db.add(news)
            total_new += 1

        db.commit()
        logger.info(f"采集完成：{source.name}，新增 {len(news_items)} 条")

    logger.info("=" * 50)
    logger.info(f"采集任务完成，共新增 {total_new} 条新闻")
    logger.info("=" * 50)

    db.close()

if __name__ == "__main__":
    run_crawl_task()
```

```python
# save as: scripts/normalize_sources.py
import openpyxl
from pathlib import Path

SOURCES_DIR = Path("E:/热点资讯/sources")

# 权重映射表
WEIGHT_MAP = {
    "政府官网": 10,
    "国际机构": 9,
    "行业协会": 8,
    "央企国企": 8,
    "行业媒体": 6,
    "一般网站": 5
}

def normalize_excel(filepath, board):
    """标准化 Excel 文件"""
    wb = openpyxl.load_workbook(filepath)
    ws = wb.active

    # 读取表头
    headers = [cell.value for cell in ws[1]]
    print(f"处理 {filepath.name}，原表头：{headers}")

    # 新表头
    new_headers = ["序号", "网站名称", "分类", "URL", "权重", "板块"]

    # 创建新工作表
    new_ws = wb.create_sheet(title="标准化", index=0)
    new_ws.append(new_headers)

    # 数据映射
    name_idx = headers.index("网站名称") if "网站名称" in headers else headers.index("信源名称")
    url_idx = headers.index("URL") if "URL" in headers else headers.index("官方网址")
    category_idx = headers.index("分类") if "分类" in headers else None

    for i, row in enumerate(ws.iter_rows(min_row=2, values_only=True), 1):
        if not row[0]:  # 跳过空行
            continue

        category = row[category_idx] if category_idx and category_idx < len(row) else "一般网站"
        weight = WEIGHT_MAP.get(category, 5)

        new_ws.append([
            i,
            row[name_idx] if name_idx < len(row) else "",
            category,
            row[url_idx] if url_idx < len(row) else "",
            weight,
            board
        ])

    # 删除原工作表
    wb.remove(ws)
    new_ws.title = "信源列表"

    wb.save(filepath)
    print(f"完成 {filepath.name}")

# 处理三个文件
normalize_excel(SOURCES_DIR / "油气行业信源.xlsx", "oil_gas")
normalize_excel(SOURCES_DIR / "海上风电信源.xlsx", "wind_power")
normalize_excel(SOURCES_DIR / "FFML 信源.xlsx", "ffml")

print("所有信源文件已标准化！")
```
